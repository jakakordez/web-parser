            function filterByDateDeko(){
            var curentUrl = window.location.href;
            var from = jQuery('#datepicker-from').val().split('.');
            var to = jQuery('#datepicker-to').val().split('.');
            //if(from.length < 3 || to.length < 3)return;
            if(from.length != 0){
               from = from.reverse().join('-');
            }

            if(to.length != 0){
               to = to.reverse().join('-');
            }



            var parts_url = curentUrl.split('?');

            var newSearchPath = [];
            var next_url = '';
            if(parts_url.length > 1){
                var url_item = parts_url[1];
                var fragments = url_item.split('&') || [];

                fragments.map(function(itm){
                    if(from !== '' && itm.indexOf('from') != -1){
                        var temp = 'from=' + from;
                        newSearchPath.push(temp);
                    }
                    else if(from !== '' && itm.indexOf('to') != -1){
                        var temp = 'to=' + to;
                        newSearchPath.push(temp);
                    }
                    else{
                        newSearchPath.push(itm);
                    }
                })


                var next_url = newSearchPath.join('&');

                if(next_url.indexOf('from=') === -1 && from !== ''){
                    next_url = next_url + '&from=' + from;
                }

                if(next_url.indexOf('to=') === -1 && to !== ''){
                    next_url = next_url + '&to=' + to;
                }


                window.location = parts_url[0] + '?' + next_url + '&a=6';
            }
        }
